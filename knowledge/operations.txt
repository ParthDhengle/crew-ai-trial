#Windows Directory
open_application | parameters: app_name | Opens an application on the desktop
create_folder | parameters: path | Creates a new directory at the specified path
delete_folder | parameters: path | Deletes a directory (and its contents if non-empty) at the specified path

#Communication
send_email | parameters: to, subject, info=None | Sends an AI-generated email to the specified recipient with optional context
send_reply_email | parameters: to, subject, retrieve_from_email=None, limit=3 | Convenience function to send contextual reply emails based on recent emails from sender
retrieveMails | parameters: from_email, limit=3 | Retrieves last N emails from a specific sender
searchMail | parameters: keyword, limit=5 | Searches emails by keyword in subject, body, or sender
send_sms | parameters: to, message | Sends an SMS to a phone number
make_call | parameters: to, message | Places a phone call and delivers a message
send_whatsapp_message | parameters: to, message | Sends a WhatsApp message to a contact

#File Management

update_file | parameters: filename, new_content | Updates content of a file
delete_file | parameters: filename | Deletes a specified file
list_files | parameters: directory | Lists all files in a directory
copy_file | parameters: source, destination | Copies a file from source to destination
move_file | parameters: source, destination | Moves a file from source to destination
search_files | parameters: directory, pattern | Searches for files matching a pattern in a directory

#Web & Search

#Calendar & Time

#Task Management

#Data Handling
